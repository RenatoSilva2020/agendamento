const ABA_AGENDAMENTOS = 'BIBLIOTECA - MANHÃ (SEDE)'; const ABA_PROFESSORES = 'PROFESSORES'; const ABA_TURMAS = 'TURMAS'; function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); // Pegar Agendamentos (ignorando a linha de cabeçalho) const sheetAgendamentos = ss.getSheetByName(ABA_AGENDAMENTOS); const agendamentos = sheetAgendamentos.getDataRange().getValues().slice(1); // Pegar Professores const sheetProfessores = ss.getSheetByName(ABA_PROFESSORES); const professores = sheetProfessores.getRange("A2:A" + sheetProfessores.getLastRow()).getValues().flat().filter(String); // Pegar Turmas const sheetTurmas = ss.getSheetByName(ABA_TURMAS); const turmas = sheetTurmas.getRange("A2:A" + sheetTurmas.getLastRow()).getValues().flat().filter(String); // Prepara o pacote de dados const dados = { agendamentos: agendamentos, professores: professores, turmas: turmas }; return ContentService.createTextOutput(JSON.stringify(dados)) .setMimeType(ContentService.MimeType.JSON); } catch(erro) { // Retorna o erro detalhado para facilitar return ContentService.createTextOutput(JSON.stringify({"erro": erro.message})) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(ABA_AGENDAMENTOS); try { const dados = JSON.parse(e.postData.contents); // Converte o array de horários selecionados em um texto (Ex: "1º, 2º") const horariosTexto = dados.horarios.join(', '); // Adiciona a linha (Data, Professor, Disciplina, Turma, Horários) sheet.appendRow([ dados.data, dados.professor, dados.disciplina, dados.turma, horariosTexto ]); return ContentService.createTextOutput(JSON.stringify({"status": "sucesso"})) .setMimeType(ContentService.MimeType.JSON); } catch (erro) { return ContentService.createTextOutput(JSON.stringify({"status": "erro", "mensagem": erro.message})) .setMimeType(ContentService.MimeType.JSON); } }
